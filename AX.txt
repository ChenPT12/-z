#!/bin/bash
# ============================================================
# perfmax.sh — Debian 13 "Trixie" Maximum Performance Tuner
# Applies system-wide: every user on the machine benefits.
# Persists across reboots via systemd unit.
# Auto-reboots at the end (15s grace window).
# ============================================================
set -euo pipefail

if [[ $EUID -ne 0 ]]; then
  echo "Run as root:  sudo bash $0"
  exit 1
fi

LOG=/var/log/perfmax.log
exec > >(tee -a "$LOG") 2>&1
echo
echo "================ PerfMax started: $(date) ================"

# ------------------------------------------------------------
# 1) Install the userspace tools we need
# ------------------------------------------------------------
echo "[1/16] Installing tooling..."
export DEBIAN_FRONTEND=noninteractive
apt-get update -y
apt-get install -y --no-install-recommends \
    linux-cpupower \
    cpufrequtils \
    irqbalance \
    power-profiles-daemon \
    sysfsutils \
    util-linux \
    ethtool \
    systemd-zram-generator || true

# ------------------------------------------------------------
# 2) Bring every CPU online (no offline cores)
# ------------------------------------------------------------
echo "[2/16] Bringing all CPU cores online..."
for c in /sys/devices/system/cpu/cpu[0-9]*/online; do
    echo 1 > "$c" 2>/dev/null || true
done

# ------------------------------------------------------------
# 3) CPU governor -> performance on every core
# ------------------------------------------------------------
echo "[3/16] Setting governor=performance on all cores..."
for g in /sys/devices/system/cpu/cpu[0-9]*/cpufreq/scaling_governor; do
    echo performance > "$g" 2>/dev/null || true
done

# ------------------------------------------------------------
# 4) Pin min freq = max freq so cores never downclock
# ------------------------------------------------------------
echo "[4/16] Pinning min freq = max freq..."
for c in /sys/devices/system/cpu/cpu[0-9]*; do
    if [[ -f "$c/cpufreq/cpuinfo_max_freq" ]]; then
        MAX=$(cat "$c/cpufreq/cpuinfo_max_freq")
        echo "$MAX" > "$c/cpufreq/scaling_max_freq" 2>/dev/null || true
        echo "$MAX" > "$c/cpufreq/scaling_min_freq" 2>/dev/null || true
    fi
done

# ------------------------------------------------------------
# 5) Enable Turbo Boost / AMD boost
# ------------------------------------------------------------
echo "[5/16] Enabling Turbo / Boost..."
[[ -f /sys/devices/system/cpu/intel_pstate/no_turbo ]] && \
    echo 0 > /sys/devices/system/cpu/intel_pstate/no_turbo
[[ -f /sys/devices/system/cpu/cpufreq/boost ]] && \
    echo 1 > /sys/devices/system/cpu/cpufreq/boost
[[ -f /sys/devices/system/cpu/amd_pstate/status ]] && \
    echo active > /sys/devices/system/cpu/amd_pstate/status 2>/dev/null || true

# ------------------------------------------------------------
# 6) Energy Performance Preference -> performance
# ------------------------------------------------------------
echo "[6/16] EPP -> performance..."
for f in /sys/devices/system/cpu/cpu[0-9]*/cpufreq/energy_performance_preference; do
    [[ -w "$f" ]] && echo performance > "$f" 2>/dev/null || true
done

# ------------------------------------------------------------
# 7) power-profiles-daemon -> performance
# ------------------------------------------------------------
echo "[7/16] power-profiles-daemon -> performance..."
if command -v powerprofilesctl >/dev/null 2>&1; then
    systemctl enable --now power-profiles-daemon || true
    powerprofilesctl set performance || true
fi

# ------------------------------------------------------------
# 8) Persist everything via a systemd unit
# ------------------------------------------------------------
echo "[8/16] Installing systemd persistence unit..."
cat > /usr/local/sbin/perfmax-apply.sh <<'EOF'
#!/bin/bash
# Re-applies max-performance settings on each boot
for c in /sys/devices/system/cpu/cpu[0-9]*/online; do
    echo 1 > "$c" 2>/dev/null || true
done
for g in /sys/devices/system/cpu/cpu[0-9]*/cpufreq/scaling_governor; do
    echo performance > "$g" 2>/dev/null || true
done
for c in /sys/devices/system/cpu/cpu[0-9]*; do
    if [[ -f "$c/cpufreq/cpuinfo_max_freq" ]]; then
        MAX=$(cat "$c/cpufreq/cpuinfo_max_freq")
        echo "$MAX" > "$c/cpufreq/scaling_max_freq" 2>/dev/null || true
        echo "$MAX" > "$c/cpufreq/scaling_min_freq" 2>/dev/null || true
    fi
done
[[ -f /sys/devices/system/cpu/intel_pstate/no_turbo ]] && \
    echo 0 > /sys/devices/system/cpu/intel_pstate/no_turbo
[[ -f /sys/devices/system/cpu/cpufreq/boost ]] && \
    echo 1 > /sys/devices/system/cpu/cpufreq/boost
for f in /sys/devices/system/cpu/cpu[0-9]*/cpufreq/energy_performance_preference; do
    [[ -w "$f" ]] && echo performance > "$f" 2>/dev/null || true
done
command -v powerprofilesctl >/dev/null 2>&1 && \
    powerprofilesctl set performance >/dev/null 2>&1 || true
exit 0
EOF
chmod +x /usr/local/sbin/perfmax-apply.sh

cat > /etc/systemd/system/perfmax.service <<'EOF'
[Unit]
Description=PerfMax — re-apply max performance settings at boot
After=multi-user.target

[Service]
Type=oneshot
ExecStart=/usr/local/sbin/perfmax-apply.sh
RemainAfterExit=yes

[Install]
WantedBy=multi-user.target
EOF
systemctl daemon-reload
systemctl enable perfmax.service

# ------------------------------------------------------------
# 9) Kernel / sysctl tuning
# ------------------------------------------------------------
echo "[9/16] Applying sysctl tuning..."
cat > /etc/sysctl.d/99-perfmax.conf <<'EOF'
# --- Memory / VM ---
vm.swappiness=10
vm.vfs_cache_pressure=50
vm.dirty_ratio=10
vm.dirty_background_ratio=5
vm.max_map_count=1048576

# --- Network throughput (BBR + fq) ---
net.core.default_qdisc=fq
net.ipv4.tcp_congestion_control=bbr
net.core.rmem_max=134217728
net.core.wmem_max=134217728
net.ipv4.tcp_rmem=4096 87380 67108864
net.ipv4.tcp_wmem=4096 65536 67108864
net.ipv4.tcp_fastopen=3
net.ipv4.tcp_mtu_probing=1
net.core.netdev_max_backlog=16384
net.core.somaxconn=8192

# --- Filesystem / inotify (helps emulators & dev tools) ---
fs.file-max=2097152
fs.inotify.max_user_watches=524288
fs.inotify.max_user_instances=8192

# --- Scheduler (Debian 13 EEVDF) ---
kernel.sched_autogroup_enabled=1
EOF
sysctl --system >/dev/null

# ------------------------------------------------------------
# 10) Optimal I/O schedulers (auto per device type)
# ------------------------------------------------------------
echo "[10/16] Installing I/O scheduler udev rule..."
cat > /etc/udev/rules.d/60-ioschedulers.rules <<'EOF'
# NVMe -> none (no overhead)
ACTION=="add|change", KERNEL=="nvme[0-9]*n[0-9]*", ATTR{queue/scheduler}="none"
# SATA SSD -> mq-deadline (low latency, TRIM-aware)
ACTION=="add|change", KERNEL=="sd[a-z]", ATTR{queue/rotational}=="0", ATTR{queue/scheduler}="mq-deadline"
# HDD -> bfq (fair, interactive-friendly)
ACTION=="add|change", KERNEL=="sd[a-z]", ATTR{queue/rotational}=="1", ATTR{queue/scheduler}="bfq"
EOF
udevadm control --reload
udevadm trigger || true

# ------------------------------------------------------------
# 11) IRQ balancing across all cores
# ------------------------------------------------------------
echo "[11/16] Enabling irqbalance..."
systemctl enable --now irqbalance || true

# ------------------------------------------------------------
# 12) Disable sleep / suspend / hibernate
# ------------------------------------------------------------
echo "[12/16] Masking sleep targets..."
systemctl mask sleep.target suspend.target hibernate.target hybrid-sleep.target || true

# ------------------------------------------------------------
# 13) ZRAM compressed swap (RAM-backed, ~3-5x faster than disk swap)
# ------------------------------------------------------------
echo "[13/16] Configuring ZRAM..."
mkdir -p /etc/systemd
cat > /etc/systemd/zram-generator.conf <<'EOF'
[zram0]
zram-size = ram / 2
compression-algorithm = zstd
swap-priority = 100
EOF
systemctl daemon-reload || true
systemctl restart systemd-zram-setup@zram0.service 2>/dev/null || true

# ------------------------------------------------------------
# 14) Raise resource limits for ALL users
# ------------------------------------------------------------
echo "[14/16] Raising limits for all users..."
cat > /etc/security/limits.d/99-perfmax.conf <<'EOF'
* soft nofile 1048576
* hard nofile 1048576
* soft nproc  unlimited
* hard nproc  unlimited
root soft nofile 1048576
root hard nofile 1048576
EOF
# systemd-managed services
mkdir -p /etc/systemd/system.conf.d
cat > /etc/systemd/system.conf.d/99-perfmax.conf <<'EOF'
[Manager]
DefaultLimitNOFILE=1048576
DefaultLimitNPROC=unlimited
EOF

# ------------------------------------------------------------
# 15) Weekly TRIM
# ------------------------------------------------------------
echo "[15/16] Enabling weekly fstrim..."
systemctl enable --now fstrim.timer || true

# ------------------------------------------------------------
# 16) Verification report
# ------------------------------------------------------------
echo
echo "================ VERIFICATION ================"
echo "Cores online: $(nproc) of $(getconf _NPROCESSORS_CONF)"
echo "Governors in use:"
cat /sys/devices/system/cpu/cpu*/cpufreq/scaling_governor 2>/dev/null | sort -u | sed 's/^/  - /'
[[ -f /sys/devices/system/cpu/intel_pstate/no_turbo ]] && \
    echo "intel no_turbo : $(cat /sys/devices/system/cpu/intel_pstate/no_turbo)  (0 = turbo ON)"
[[ -f /sys/devices/system/cpu/cpufreq/boost ]] && \
    echo "cpufreq boost  : $(cat /sys/devices/system/cpu/cpufreq/boost)  (1 = ON)"
if command -v powerprofilesctl >/dev/null 2>&1; then
    echo "PPD profile    : $(powerprofilesctl get)"
fi
echo "TCP congestion : $(sysctl -n net.ipv4.tcp_congestion_control)"
echo "Default qdisc  : $(sysctl -n net.core.default_qdisc)"
echo "Swappiness     : $(sysctl -n vm.swappiness)"
echo "==============================================="
echo "Log saved to: $LOG"
echo
echo "Rebooting in 15 seconds. Press Ctrl+C to cancel."
sleep 15
systemctl reboot
